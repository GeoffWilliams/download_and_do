Overview

Download_and_do::Run

Download a file, extract it and then run a command.

Perfect for things like downloadable tarballs containing RPMs to install. The downloaded archive will be retained after running. Use the checksum to force a local update and reinstall

Parameters:

  • source (Any) —

    Location to download archive from

  • download_dir (Any) (defaults to: $download_and_do::download_dir) —

    Directory to download archive to if not using module default

  • local_file (Any) (defaults to: $title) —

    Where to save the file (inside $download_dir)

  • extract_dir (Any) (defaults to: $download_and_do::extract_dir) —

    Directory to extract archive to

  • checksum (Any) (defaults to: undef) —

    Checksum of local archive file to see if update needed (SHA1)

  • run_relative (Any) —

    Commands to run (relative to Extract dir) after downloading

  • path (Any) (defaults to: $download_and_do::path) —

    PATH (array) to use when running command after extraction

  • creates (Any) (defaults to: undef) —

    Presence of this file indicates we do not need to download or execute anything

  • unless (Any) (defaults to: undef) —

    Pass-through parameter to exec to control whether to run or not

  • onlyif (Any) (defaults to: undef) —

    Pass-through parameter to exec to control whether to run or not

  • user (Any) (defaults to: undef) —

    User for file ownership and to run command with

  • group (Any) (defaults to: undef) —

    Group for file ownership

  • environment (Any) (defaults to: undef) —

    Environment variables (BASH) to run command with

  • allow_insecure (Any) (defaults to: $download_and_do::allow_insecure) —

    Allow insecure HTTPS downloads

define download_and_do::extract_and_run(
    $source,
    $run_relative,
    $local_file     = $title,
    $download_dir   = $download_and_do::download_dir,
    $extract_dir    = $download_and_do::extract_dir,
    $checksum       = undef,
    $path           = $download_and_do::path,
    $creates        = undef,
    $unless         = undef,
    $onlyif         = undef,
    $user           = undef,
    $group          = undef,
    $environment    = undef,
    $allow_insecure = $download_and_do::allow_insecure,
) {

  $run_relative_exec  = "run_relative after install ${title}"
  $local_file_path    = "${download_dir}/${local_file}"

  if $checksum {
    $checksum_type = "sha1"
  } else {
    $checksum_type = undef
  }

  archive { $local_file_path:
    ensure         => present,
    extract        => true,
    extract_path   => $extract_dir,
    source         => $source,
    checksum       => $checksum,
    checksum_type  => $checksum_type,
    cleanup        => true,
    creates        => $creates,
    user           => $user,
    group          => $group,
    allow_insecure => $allow_insecure,
    notify         => Exec[$run_relative_exec]
  }


  exec { $run_relative_exec:
    command     => "cd ${extract_dir} && ${run_relative}",
    refreshonly => true,
    creates     => $creates,
    unless      => $unless,
    onlyif      => $onlyif,
    path        => $path,
    user        => $user,
    environment => $environment,
  }
}
