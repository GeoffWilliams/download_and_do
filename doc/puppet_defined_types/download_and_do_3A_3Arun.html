Download_and_do::Run

Download a file and then run it. Perfect for things like self-installing bash scripts. The script will be retained after running. Use the checksum to force a local update and reinstall

Parameters:

  • source (Any) —

    File to download

  • local_file (Any) (defaults to: $title) —

    Where to save the file (inside $download_dir)

  • download_dir (Any) (defaults to: $download_and_do::download_dir) —

    Drectory to download to instead of module default

  • path (Any) (defaults to: $download_and_do::path) —

    PATH to use for executing file

  • checksum (Any) (defaults to: undef) —

    Checksum (SHA1) of file, used to detect updates

  • creates (Any) (defaults to: undef) —

    Presence of this file indicates we do not need to download or execute anything

  • unless (Any) (defaults to: undef) —

    Pass-through parameter to exec to control whether to run or not

  • onlyif (Any) (defaults to: undef) —

    Pass-through parameter to exec to control whether to run or not

  • user (Any) (defaults to: undef) —

    User for file ownership and to run command with

  • group (Any) (defaults to: undef) —

    Group for file ownership

  • environment (Any) (defaults to: undef) —

    Environment variables (BASH) to run command with

  • arguments (Any) (defaults to: "") —

    Arguments to pass to the downloaded file to be run

  • allow_insecure (Any) (defaults to: $download_and_do::allow_insecure) —

    Allow insecure HTTPS downloads

  • provider (Any) (defaults to: $download_and_do::provider) —

    Set the exec provider - needed if cd is a builtin

  • timeout (Any) (defaults to: undef) —

    How long to wait before killing command we were told to run, in seconds. Pass 0 to wait upto forever seconds for the command to complete

  • proxy_server (Any) (defaults to: undef) —

    specify a proxy server, with port number if needed. ie: example.com:8080.

  • proxy_type (Any) (defaults to: undef) —

    proxy server type (none|http|https|ftp)

define download_and_do::run(
  $source,
  $local_file     = $title,
  $download_dir   = $download_and_do::download_dir,
  $path           = $download_and_do::path,
  $checksum       = undef,
  $creates        = undef,
  $unless         = undef,
  $onlyif         = undef,
  $user           = undef,
  $group          = undef,
  $environment    = undef,
  $arguments      = "",
  $allow_insecure = $download_and_do::allow_insecure,
  $provider       = $download_and_do::provider,
  $timeout        = undef,
  $proxy_server   = undef,
  $proxy_type     = undef,
) {

  $chmod_file_exec  = "chmod after download ${title}"
  $run_file_exec    = "run after download ${title}"
  $local_file_path  = "${download_dir}/${local_file}"

  if $checksum {
    $checksum_type = "sha1"
  } else {
    $checksum_type = undef
  }

  # Download will always be owned by root, user parameter only for extract
  archive { $local_file_path:
    ensure         => present,
    extract        => false,
    path           => $local_file_path,
    source         => $source,
    checksum       => $checksum,
    checksum_type  => $checksum_type,
    creates        => $creates,
    group          => $group,
    allow_insecure => $allow_insecure,
    proxy_server   => $proxy_server,
    proxy_type     => $proxy_type,
    notify         => Exec[$chmod_file_exec]
  }

  # chmod +x the script so it can be run natively.  Separate exec due to being
  # done as root always
  exec { $chmod_file_exec:
    command     => "/bin/chmod +x ${local_file_path}",
    refreshonly => true,
    user        => "root",
    notify      => Exec[$run_file_exec],
  }

  # Run the now-executable script if we need to
  exec { $run_file_exec:
    command     => "${local_file_path} ${arguments}",
    refreshonly => true,
    path        => $path,
    creates     => $creates,
    unless      => $unless,
    onlyif      => $onlyif,
    user        => $user,
    environment => $environment,
    provider    => $provider,
    timeout     => $timeout,
  }

}
